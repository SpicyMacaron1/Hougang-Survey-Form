📋 Step-by-step setup guide
1
Create a Google Sheet
Go to sheets.google.com and create a new blank spreadsheet. Name it anything you like (e.g. Hougang Survey 2026).
2
Open Apps Script
In your sheet, click Extensions → Apps Script. Delete all existing placeholder code in the editor.
3
Paste the script below
Copy the full script and paste it into the editor. Click Save project (💾 icon).
4
Deploy as Web App
Click Deploy → New deployment. For type select Web app. Set Execute as: Me and Who has access: Anyone. Click Deploy. Authorise the permissions when prompted.
5
Copy the Web App URL
After deployment, copy the URL (starts with https://script.google.com/macros/s/…/exec). Paste it above and click Save & Test.
Google Apps Script — paste this into your script editor 
const SHEET_NAME = "Responses";

const HEADERS = [
  "Timestamp","Name","Phone","Age Range","Gender",
  "Years in Hougang","Housing Type",
  "ECO: Cost of Living (1-5)","ECO: Job Opportunities (1-5)",
  "ECO: Local Shopping Frequency","ECO: Retail Amenities (1-5)",
  "ECO: Open Feedback",
  "ENV: Cleanliness (1-5)","ENV: Green Spaces (1-5)",
  "ENV: Eco Practices","ENV: Noise & Air Quality",
  "ENV: Open Feedback",
  "SOC: Safety (1-5)","SOC: Community Spirit (1-5)",
  "SOC: Healthcare Satisfaction","SOC: Community Participation",
  "SOC: Education (1-5)","SOC: Values Open","SOC: Improvements Open",
  "Overall Satisfaction (1-5)","Would Recommend"
];

function doPost(e) {
  const lock = LockService.getScriptLock();
  lock.tryLock(10000);
  try {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    let sheet = ss.getSheetByName(SHEET_NAME);
    if (!sheet) {
      sheet = ss.insertSheet(SHEET_NAME);
      sheet.appendRow(HEADERS);
      const hdr = sheet.getRange(1,1,1,HEADERS.length);
      hdr.setBackground("#1F4E79").setFontColor("#ffffff")
         .setFontWeight("bold").setFontSize(10);
      sheet.setFrozenRows(1);
      sheet.setColumnWidth(1, 160);
    }
    const d = JSON.parse(e.postData.contents);
    sheet.appendRow([
      d.timestamp, d.name, d.phone, d.age, d.gender,
      d.duration, d.housing,
      d.eco_cost_of_living, d.eco_job_opportunities,
      d.eco_local_shopping, d.eco_retail_amenities, d.eco_feedback,
      d.env_cleanliness, d.env_green_spaces,
      d.env_eco_practices, d.env_noise_air, d.env_challenge,
      d.soc_safety, d.soc_community, d.soc_healthcare,
      d.soc_participation, d.soc_education,
      d.soc_values, d.soc_improvements,
      d.overall_satisfaction, d.recommend
    ]);
    return ContentService
      .createTextOutput(JSON.stringify({result:"success"}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({result:"error",error:err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  } finally {
    lock.releaseLock();
  }
}

function doGet(e) {
  return ContentService
    .createTextOutput(JSON.stringify({result:"ok",message:"Endpoint active"}))
    .setMimeType(ContentService.MimeType.JSON);
}
